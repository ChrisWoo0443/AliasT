    ```zsh
    # ── 6b. Accept keybindings ─────────────────────────────────────────
    _alias_accept_suggestion() {
      if [[ -n "$POSTDISPLAY" ]]; then
        BUFFER="${BUFFER}${POSTDISPLAY}"
        CURSOR=$#BUFFER
        _alias_clear_ghost
      else
        zle expand-or-complete
      fi
    }
    zle -N _alias_accept_suggestion
    bindkey '^I' _alias_accept_suggestion  # ^I is Tab
    ```

    Key details per Research Pattern 3:
    - Check POSTDISPLAY first. If ghost text exists, append it to BUFFER, move CURSOR to end, clear ghost text.
    - If no ghost text, fall through to `expand-or-complete` (native Tab behavior). This prevents breaking native completion, fzf-tab, etc. (per Pitfall 3).
    - Do NOT save/restore the original binding with `zle -A` -- that adds complexity for no gain since `expand-or-complete` is the standard target.

    **Feature 2: Shift+Tab accepts word-by-word (D-08, SUGG-03).**
    Add right after the Tab binding:

    ```zsh
    _alias_accept_word() {
      if [[ -n "$POSTDISPLAY" ]]; then
        local remaining="$POSTDISPLAY"
        local word=""
        # Grab leading whitespace
        if [[ "$remaining" =~ '^[[:space:]]*' ]]; then
          word="${MATCH}"
          remaining="${remaining#$MATCH}"
        fi
        # Grab next non-space word
        if [[ "$remaining" =~ '^[^[:space:]]+' ]]; then
          word="${word}${MATCH}"
        fi
        BUFFER="${BUFFER}${word}"
        CURSOR=$#BUFFER
        _alias_clear_ghost
        # Re-request suggestion for updated buffer
        _alias_request_suggestion
      else
        zle .complete-word
      fi
    }
    zle -N _alias_accept_word
    bindkey '^[[Z' _alias_accept_word  # ^[[Z is Shift+Tab
    ```

    Key details per Research Pattern 4:
    - Extract leading whitespace + next word from POSTDISPLAY.
    - Append to BUFFER, update CURSOR.
    - Clear ghost text, then re-request a suggestion for the updated buffer (the remaining text should appear as the new ghost text).
    - Fall through to `.complete-word` if no ghost text.

    **Feature 3: precmd hook records commands (D-03).**
    Add as a new section before Section 8 (compatibility detection):

    ```zsh
    # ── 7b. Command recording (precmd hook) ────────────────────────────
    typeset -g _ALIAS_LAST_HISTNUM=0

    _alias_precmd_record() {
      # Get the last history entry number and command
      local hist_line
      hist_line="$(fc -l -1 2>/dev/null)" || return
      local histnum="${hist_line%%[[:space:]]*}"

      # Skip if we already recorded this history event (per Pitfall 7)
      [[ "$histnum" == "$_ALIAS_LAST_HISTNUM" ]] && return
      _ALIAS_LAST_HISTNUM="$histnum"

      # Extract command text (strip leading number + spaces from fc output)
      local cmd="${hist_line#*[0-9]  }"
      [[ -z "$cmd" ]] && return

      _alias_connect || return

      # Escape for JSON
      local escaped_cmd="${cmd//\\/\\\\}"
      escaped_cmd="${escaped_cmd//\"/\\\"}"
      local escaped_cwd="${PWD//\\/\\\\}"
      escaped_cwd="${escaped_cwd//\"/\\\"}"

      (( _ALIAS_REQ_ID++ ))
      local msg="{\"id\":\"r${_ALIAS_REQ_ID}\",\"type\":\"record\",\"cmd\":\"${escaped_cmd}\",\"cwd\":\"${escaped_cwd}\"}"

      # Fire-and-forget: send but don't wait for response
      print -u $_ALIAS_FD "$msg" 2>/dev/null || _alias_reconnect
    }

    autoload -Uz add-zsh-hook
    add-zsh-hook precmd _alias_precmd_record
    ```

    Key details per Research and Pitfall 7:
    - Track `_ALIAS_LAST_HISTNUM` to avoid sending the same command twice on empty Enter presses.
    - Use `fc -l -1` to get the last history entry. Parse out the history number and command text.
    - Fire-and-forget: send the Record message but do not read the Ack response. This keeps precmd fast.
    - JSON-escape backslashes and double quotes in both cmd and cwd.
