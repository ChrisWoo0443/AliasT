    **Add migration function to main.rs** (before `init_tracing`):

    ```rust
    fn migrate_data_directory() -> Result<()> {
        let old_data_dir = directories::BaseDirs::new()
            .map(|dirs| dirs.data_local_dir().join("alias"))
            .unwrap_or_else(|| {
                let home = std::env::var("HOME").unwrap_or_else(|_| "/tmp".to_string());
                PathBuf::from(home).join(".local").join("share").join("alias")
            });

        let new_data_dir = directories::BaseDirs::new()
            .map(|dirs| dirs.data_local_dir().join("aliast"))
            .unwrap_or_else(|| {
                let home = std::env::var("HOME").unwrap_or_else(|_| "/tmp".to_string());
                PathBuf::from(home).join(".local").join("share").join("aliast")
            });

        // Migrate history.db (per D-02: move, not copy, silent)
        let old_db = old_data_dir.join("history.db");
        let new_db = new_data_dir.join("history.db");
        if old_db.exists() && !new_db.exists() {
            std::fs::create_dir_all(&new_data_dir)?;
            std::fs::rename(&old_db, &new_db)?;
        }

        // Migrate daemon.log (per D-03: same logic)
        let old_log = old_data_dir.join("daemon.log");
        let new_log = new_data_dir.join("daemon.log");
        if old_log.exists() && !new_log.exists() {
            std::fs::create_dir_all(&new_data_dir)?;
            std::fs::rename(&old_log, &new_log)?;
        }

        Ok(())
    }
    ```
