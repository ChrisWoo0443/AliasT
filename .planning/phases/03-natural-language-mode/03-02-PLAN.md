    ```rust
    #[tokio::test]
    async fn test_generate_returns_command() {
        let mock = Arc::new(MockAiBackend {
            response: "ls -la".to_string(),
        });
        let (socket_path, cancel_token, _tmp) = spawn_daemon_with_ai(mock).await;
        let stream = connect(&socket_path).await;
        let (reader, mut writer) = stream.into_split();
        let mut buf_reader = BufReader::new(reader);
        
        // Send generate request
        let request = r#"{"type":"generate","id":"r1","prompt":"list files"}"#;
        writer.write_all(format!("{}\n", request).as_bytes()).await.unwrap();
        
        let mut response_line = String::new();
        buf_reader.read_line(&mut response_line).await.unwrap();
        
        let response: Response = serde_json::from_str(response_line.trim()).unwrap();
        match response {
            Response::Command { id, text } => {
                assert_eq!(id, "r1");
                assert_eq!(text, "ls -la");
            }
            other => panic!("Expected Command response, got: {:?}", other),
        }
        
        cancel_token.cancel();
    }
    ```
